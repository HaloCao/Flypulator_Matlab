1function q   = DCM2Quat(R)
 2
 3%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 4%
 5% M-file: Tool_DCM2Quat
 6%         -> impelmeted as M-Function
 7% Author: Marcel Tkocz
 8% Version: 27.01.2008
 9%
10% Description:
11%   This Function converts roll-pitch-yaw angels into their quaternion
12%   equivalents.
13%
14% Inputs:
15%   Orientation ... orientation in roll-pitch-yaw angels
16%
17% Outputs:
18%   q           ... orientation in quaternions
19%   
20%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
21
22%--------------------------------------------------------------------------
23% Calculation of Quaternions based on q1
24%--------------------------------------------------------------------------
25
26q1 = 1/2*sqrt(1 + R(1,1) - R(2,2) - R(3,3));
27q2 = 1/(4*q1)*( R(1,2) + R(2,1));
28q3 = 1/(4*q1)*( R(1,3) + R(3,1));
29q4 = 1/(4*q1)*(-R(2,3) + R(3,2));
30
31if round(q1) ~= 0
32    q = [q1; q2; q3; q4];
33end
34
35%--------------------------------------------------------------------------
36% Calculation of Quaternions based on q2
37%--------------------------------------------------------------------------
38
39q2 = 1/2*sqrt(1 - R(1,1) + R(2,2) - R(3,3));
40q1 = 1/(4*q2)*( R(1,2) + R(2,1));
41q3 = 1/(4*q2)*( R(2,3) + R(3,2));
42q4 = 1/(4*q2)*( R(1,3) - R(3,1));
43
44if round(q2) ~= 0
45    q = [q1; q2; q3; q4];
46end
47
48%--------------------------------------------------------------------------
49% Calculation of Quaternions based on q3
50%--------------------------------------------------------------------------
51
52q3 = 1/2*sqrt(1 - R(1,1) - R(2,2) + R(3,3));
53q1 = 1/(4*q3)*( R(1,3) + R(3,1));
54q2 = 1/(4*q3)*( R(2,3) + R(3,2));
55q4 = 1/(4*q3)*(-R(1,2) + R(2,1));
56
57if round(q3) ~= 0
58    q = [q1; q2; q3; q4];
59end
60
61%--------------------------------------------------------------------------
62% Calculation of Quaternions based on q4
63%--------------------------------------------------------------------------
64
65q4 = 1/2*sqrt(1 + R(1,1) + R(2,2) + R(3,3));
66q1 = 1/(4*q4)*(-R(2,3) + R(3,2));
67q2 = 1/(4*q4)*(-R(3,1) + R(1,3));
68q3 = 1/(4*q4)*(-R(1,2) + R(2,1));
69    
70if round(q4) ~= 0
71    q = [q1; q2; q3; q4];
72end
73
74end
